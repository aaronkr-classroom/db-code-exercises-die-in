1. 학사 관리 DB를 구축하기 위해 다음과 같은 릴레이션 스키마와 무결성 제약조건을 정의하였다. 이를 기초로 하여 내부 스키마를 정의하시오.
> 정답:
> CREATE TABLE 학과 (학과id VARCHAR(20) PRIMARY KEY, 학과명 VARCHAR(50) NOT NULL UNIQUE, 학과사무실 VARCHAR(50), 전화번호 VARCHAR(20) UNIQUE); \
> CREATE TABLE 교수 (교수id VARCHAR(20) PRIMARY KEY, 이름 VARCHAR(20) NOT NULL, 전공 VARCHAR(50) NOT NULL, 학과id VARCHAR(20) NOT NULL, FOREIGN KEY (학과id) REFERENCES 학과(학과id)); \
> CREATE TABLE 학생 (학번 VARCHAR(20) PRIMARY KEY, 학과id VARCHAR(20), 이름 VARCHAR(20) NOT NULL, 주소 VARCHAR(100), 학년 INT NOT NULL CHECK(학년 BETWEEN 1 AND 4), 지도교수 VARCHAR(20), FOREIGN KEY (학과id) REFERENCES 학과(학과id), FOREIGN KEY (지도교수) REFERENCES 교수(교수id)); \
> CREATE TABLE 교과목 (교과목번호 VARCHAR(20) PRIMARY KEY, 교과목명 VARCHAR(50) NOT NULL, 학점 INT DEFAULT 3 NOT NULL, 담당교수 VARCHAR(20), FOREIGN KEY (담당교수) REFERENCES 교수(교수id)); \
> CREATE TABLE 수강하다 (학번 VARCHAR(20), 교과목번호 VARCHAR(20), 성적 VARCHAR(2) CHECK(성적 IN ('A', 'B', 'C', 'D', 'F')), PRIMARY KEY (학번, 교과목번호), FOREIGN KEY (학번) REFERENCES 학생(학번), FOREIGN KEY (교과목번호) REFERENCES 교과목(교과목번호));

2. 국내 관광을 전문으로 하는 한 여행사의 DB를 구축하기 위해 다음과 같은 릴레이션 스키마와 무결성 제약조건을 정의하였다. 이를 기초로 하여 내부 스키마를 정의하고, 임의로 2개의 인덱스를 정의하시오.
> 정답:
> CREATE TABLE 고객등급코드 (코드 INT PRIMARY KEY, 등급 VARCHAR(20), 기준 VARCHAR(50)); \
> CREATE TABLE 직원업무코드 (코드 INT PRIMARY KEY, 업무 VARCHAR(50)); \
> CREATE TABLE 고객 (id VARCHAR(20) PRIMARY KEY, 이름 VARCHAR(20) NOT NULL, 휴대폰 VARCHAR(20) NOT NULL UNIQUE, 주소 VARCHAR(100), 등급코드 INT DEFAULT 3); \
> CREATE TABLE 직원 (사번 VARCHAR(20) PRIMARY KEY, 이름 VARCHAR(20) NOT NULL, 생년월일 DATE NOT NULL UNIQUE, 연락처 VARCHAR(20), 급여 INT, 담당업무 VARCHAR(50) NOT NULL, 입사일 DATE NOT NULL); \
> CREATE TABLE 여행상품 (여행번호 VARCHAR(20) PRIMARY KEY, 출발지 VARCHAR(50) NOT NULL, 도착지 VARCHAR(50) NOT NULL, 프로그램 VARCHAR(100), 시작일시 DATE NOT NULL, 종료일시 DATE NOT NULL, 최소출발인원 INT, 최대인원 INT, 여행경비 INT NOT NULL, 예약금 INT, 출발여부 CHAR(1) DEFAULT 'N', 담당직원사번 VARCHAR(20), FOREIGN KEY (담당직원사번) REFERENCES 직원(사번)); \
> CREATE TABLE 관광버스 (차량번호 VARCHAR(20) PRIMARY KEY, 좌석수 INT, 출고년도 INT); \
> CREATE TABLE 운전기사 (기사_id VARCHAR(20) PRIMARY KEY, 이름 VARCHAR(20) NOT NULL, 생년월일 DATE NOT NULL, 휴대폰 VARCHAR(20) NOT NULL, 시급 INT DEFAULT 15000, 계약일 DATE, 계약기간 INT); \
> CREATE TABLE 예약하다 (고객_id VARCHAR(20), 여행번호 VARCHAR(20), 예약일자 DATE DEFAULT SYSDATE, 예약금결제여부 CHAR(1) DEFAULT 'N', 여행경비결제여부 CHAR(1) DEFAULT 'N', PRIMARY KEY(고객_id, 여행번호), FOREIGN KEY(고객_id) REFERENCES 고객(id), FOREIGN KEY(여행번호) REFERENCES 여행상품(여행번호)); \
> CREATE TABLE 기사배정하다 (여행번호 VARCHAR(20), 기사_id VARCHAR(20), 근무시간 INT, PRIMARY KEY(여행번호, 기사_id), FOREIGN KEY(여행번호) REFERENCES 여행상품(여행번호), FOREIGN KEY(기사_id) REFERENCES 운전기사(기사_id)); \
> CREATE TABLE 차량배정하다 (여행번호 VARCHAR(20), 차량번호 VARCHAR(20), PRIMARY KEY(여행번호, 차량번호), FOREIGN KEY(여행번호) REFERENCES 여행상품(여행번호), FOREIGN KEY(차량번호) REFERENCES 관광버스(차량번호)); \
> CREATE INDEX idx_고객_이름 ON 고객(이름); \
> CREATE INDEX idx_여행상품_출발지 ON 여행상품(출발지);
